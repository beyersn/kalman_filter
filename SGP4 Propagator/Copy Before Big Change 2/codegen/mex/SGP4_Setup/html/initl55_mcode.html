1% -----------------------------------------------------------------------------
  2%
  3%                            procedure initl
  4%
  5%   this procedure initializes the spg4 propagator. all the initialization is
  6%     consolidated here instead of having multiple loops inside other routines.
  7%
  8% Author: 
  9%   Jeff Beck 
 10%   beckja@alumni.lehigh.edu
 11%   1.0 (aug 7, 2006) - update for paper dav
 12%   1.1 nov 16, 2007 - update for better compliance
 13% original comments from Vallado C++ version:
 14%   author        : david vallado                  719-573-2600   28 jun 2005
 15%
 16%   inputs        :
 17%     satrec.ecco        - eccentricity                           0.0 - 1.0
 18%     epoch       - epoch time in days from jan 0, 1950. 0 hr
 19%     satrec.inclo       - inclination of satellite
 20%     satrec.no          - mean motion of satellite
 21%     satrec.satnum        - satellite number
 22%
 23%   outputs       :
 24%     ainv        - 1.0 / a
 25%     ao          - semi major axis
 26%     satrec.con41       -
 27%     con42       - 1.0 - 5.0 cos(i)
 28%     cosio       - cosine of inclination
 29%     cosio2      - cosio squared
 30%     einv        - 1.0 / e
 31%     eccsq       - eccentricity squared
 32%     satrec.method      - flag for deep space                    'd', 'n'
 33%     omeosq      - 1.0 - satrec.ecco * satrec.ecco
 34%     posq        - semi-parameter squared
 35%     rp          - radius of perigee
 36%     rteosq      - square root of (1.0 - satrec.ecco*satrec.ecco)
 37%     sinio       - sine of inclination
 38%     satrec.gsto        - gst at time of observation               rad
 39%     satrec.no          - mean motion of satellite
 40%
 41%   locals        :
 42%     ak          -
 43%     d1          -
 44%     del         -
 45%     adel        -
 46%     po          -
 47%
 48%   coupling      :
 49%     gstime      - find greenwich sidereal time from the julian date
 50%
 51%   references    :
 52%     hoots, roehrich, norad spacetrack report #3 1980
 53%     hoots, norad spacetrack report #6 1986
 54%     hoots, schumacher and glover 2004
 55%     vallado, crawford, hujsak, kelso  2006
 56%  ----------------------------------------------------------------------------*/
 57
 58function [ainv,  ao, con42, cosio, cosio2, einv, eccsq, omeosq, posq, rp,...
 59    rteosq, sinio] = initl(epoch);
 60
 61   % /* -------------------- wgs-72 earth constants ----------------- */
 62   %     // sgp4fix identify constants and allow alternate values
 63   x2o3   = 2.0 / 3.0;
 64   global opsmode
 65   global satrec gravc
 66
 67   % /* ------------- calculate auxillary epoch quantities ---------- */
 68   eccsq  = satrec.ecco * satrec.ecco;
 69   omeosq = 1.0 - eccsq;
 70   rteosq = sqrt(omeosq);
 71   cosio  = cos(satrec.inclo);
 72   cosio2 = cosio * cosio;
 73
 74   % /* ------------------ un-kozai the mean motion ----------------- */
 75   ak    = (gravc.xke / satrec.no)^x2o3;
 76   d1    = 0.75 * gravc.j2 * (3.0 * cosio2 - 1.0) / (rteosq * omeosq);
 77   del   = d1 / (ak * ak);
 78   adel  = ak * (1.0 - del * del - del *...
 79       (1.0 / 3.0 + 134.0 * del * del / 81.0));
 80   del   = d1/(adel * adel);
 81   satrec.no    = satrec.no / (1.0 + del);
 82
 83   ao    = (gravc.xke / satrec.no)^x2o3;
 84   sinio = sin(satrec.inclo);
 85   po    = ao * omeosq;
 86   con42 = 1.0 - 5.0 * cosio2;
 87   satrec.con41 = -con42-cosio2-cosio2;
 88   ainv  = 1.0 / ao;
 89   einv  = 1.0 / satrec.ecco;
 90   posq  = po * po;
 91   rp    = ao * (1.0 - satrec.ecco);
 92   satrec.method = 'n';
 93
 94   % sgp4fix modern approach to finding sidereal time
 95   satrec.gsto = gstime(satrec.jdsatepoch);
 96
 97%    global idebug dbgfile
 98%    if isempty(idebug)
 99%        idebug = 0;
100%    elseif idebug
101%        debug5;
102%    end
103
104   return;
