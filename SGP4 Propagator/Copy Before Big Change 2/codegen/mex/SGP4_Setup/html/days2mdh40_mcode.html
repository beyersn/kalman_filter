1% ------------------------------------------------------------------------------
 2%
 3%                           function days2mdh
 4%
 5%  this function converts the day of the year, days, to the equivalent month
 6%    day, hour, minute and second.
 7%
 8%  author        : david vallado                  719-573-2600   22 jun 2002
 9%
10%  revisions
11%                -
12%
13%  inputs          description                    range / units
14%    year        - year                           1900 .. 2100
15%    days        - julian day of the year         0.0  .. 366.0
16%
17%  outputs       :
18%    mon         - month                          1 .. 12
19%    day         - day                            1 .. 28,29,30,31
20%    hr          - hour                           0 .. 23
21%    minute      - minute                         0 .. 59
22%    sec         - second                         0.0 .. 59.999
23%
24%  locals        :
25%    dayofyr     - day of year
26%    temp        - temporary extended values
27%    inttemp     - temporary integer value
28%    i           - index
29%    lmonth(12)  - integer array containing the number of days per month
30%
31%  coupling      :
32%    none.
33%
34% [mon,day,hr,minute,sec] = days2mdh ( year,days);
35% -----------------------------------------------------------------------------
36
37function [mon,day,hr,minute,sec] = days2mdh ( year,days)
38
39        % Set up array of days in month
40        lmonth = 31*zeros(12,1);
41        for i= 1 : 12
42            if i == 2
43                lmonth(i,1)= 28;
44            end;
45            if i == 4 || i == 6 || i == 9 || i == 11
46                lmonth(i,1)= 30;
47            end;
48        end
49
50        dayofyr= floor(days );
51
52        % Find month and day of month
53        if rem(year-1900,4) == 0
54            lmonth(2)= 29; % Leap year
55        end
56
57        i= 1;
58        inttemp= 0;
59        while ( dayofyr > inttemp + lmonth(i) ) && ( i < 12 )
60            inttemp= inttemp + lmonth(i);
61            i= i+1;
62        end
63
64        mon= i;
65        day= dayofyr - inttemp;
66
67        % Find hours minutes and seconds
68        temp= (days - dayofyr )*24.0;
69        hr  = fix( temp );
70        temp= (temp-hr) * 60.0;
71        minute = fix( temp );
72        sec = (temp-minute) * 60.0;
73
