1function Dipole_Command_s = BDot_Controller(B_s, Command_Trigger, Ts)
 2%#codegen
 3% BDot_Controller
 4% Author: Brandon Jackson
 5% Contact: bajackso@mtu.edu
 6% Date: 7 August 2013
 7%
 8% This function is a Bdot based detumble controller.
 9
10persistent Bint Bdot
11Dipole_Gain = -5E6;
12
13if Command_Trigger == 1 || isempty(Bdot);
14    % This is the measurement stage of the controller,
15    if isempty(Bint)
16        Bint = zeros(3,1);
17    else
18        % Do nothing
19    end
20    
21    Bdot = 0.8*(B_s - Bint);
22    Bint = Bdot*Ts + Bint;
23    Dipole_Command_s = [0.0; 0.0; 0.0];
24    
25else
26    
27    Dipole_Command_s = Dipole_Gain*diag([4.3 4.3 3.9])*Bdot;
28    
29    
30end
