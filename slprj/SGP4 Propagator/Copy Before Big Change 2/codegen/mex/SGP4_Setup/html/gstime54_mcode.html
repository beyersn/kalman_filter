1% -----------------------------------------------------------------------------
 2%
 3%                           function gstime
 4%
 5%  this function finds the greenwich sidereal time (iau-82).
 6%
 7%  author        : david vallado                  719-573-2600    7 jun 2002
 8%
 9%  revisions
10%                -
11%
12%  inputs          description                    range / units
13%    jdut1       - julian date of ut1             days from 4713 bc
14%
15%  outputs       :
16%    gst         - greenwich sidereal time        0 to 2pi rad
17%
18%  locals        :
19%    temp        - temporary variable for reals   rad
20%    tut1        - julian centuries from the
21%                  jan 1, 2000 12 h epoch (ut1)
22%
23%  coupling      :
24%
25%  references    :
26%    vallado       2007, 193, Eq 3-43
27%
28% gst = gstime(jdut1);
29% -----------------------------------------------------------------------------
30
31function gst = gstime(jdut1);
32
33        twopi      = 2.0*pi;
34        deg2rad    = pi/180.0;
35
36        % ------------------------  implementation   ------------------
37        tut1= ( jdut1 - 2451545.0 ) / 36525.0;
38
39        temp = - 6.2e-6 * tut1 * tut1 * tut1 + 0.093104 * tut1 * tut1  ...
40               + (876600.0 * 3600.0 + 8640184.812866) * tut1 + 67310.54841;
41
42        % 360/86400 = 1/240, to deg, to rad
43        temp = rem( temp*deg2rad/240.0,twopi );
44
45        % ------------------------ check quadrants --------------------
46        if ( temp < 0.0 )
47            temp = temp + twopi;
48          end
49
50        gst = temp;
51
