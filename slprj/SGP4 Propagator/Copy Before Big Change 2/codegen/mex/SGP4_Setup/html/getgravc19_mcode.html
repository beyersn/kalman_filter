1% -----------------------------------------------------------------------------
 2%
 3%                           function getgravc
 4%
 5%  this function gets constants for the propagator. note that gravc.mu is identified to
 6%    facilitiate comparisons with newer models.
 7%
 8%  author        : david vallado                  719-573-2600   21 jul 2006
 9%
10%  inputs        :
11%    whichconst  - which set of constants to use  721, 72, 84
12%
13%  outputs       :
14%    tumin       - minutes in one time unit
15%    mu          - earth gravitational parameter
16%    radiusearthkm - radius of the earth in km
17%    xke         - reciprocal of gravc.tumin
18%    j2, j3, j4  - un-normalized zonal harmonic values
19%    j3oj2       - j3 divided by j2
20%
21%  locals        :
22%
23%  coupling      :
24%
25%  references    :
26%    norad spacetrack report #3
27%    vallado, crawford, hujsak, kelso  2006
28% [gravc.tumin, gravc.mu, gravc.radiusearthkm, gravc.xke, j2, j3, j4, j3oj2] = getgravc(whichconst);
29%  --------------------------------------------------------------------------- */
30
31function getgravc(whichconst);
32
33    global gravc
34      switch whichconst
35          case 721
36           % -- wgs-72 low precision str#3 constants --
37           gravc.mu     = 398600.79964;        %// in km3 / s2
38           gravc.radiusearthkm = 6378.135;     %// km
39           gravc.xke    = 0.0743669161;
40           gravc.tumin  = 1.0 / gravc.xke;
41           gravc.j2     =   0.001082616;
42           gravc.j3     =  -0.00000253881;
43           gravc.j4     =  -0.00000165597;
44           gravc.j3oj2  =  gravc.j3 / gravc.j2;
45          case 72
46           % ------------ wgs-72 constants ------------
47           gravc.mu     = 398600.8;            %// in km3 / s2
48           gravc.radiusearthkm = 6378.135;     %// km
49           gravc.xke    = 60.0 / sqrt(gravc.radiusearthkm*gravc.radiusearthkm*gravc.radiusearthkm/gravc.mu);
50           gravc.tumin  = 1.0 / gravc.xke;
51           gravc.j2     =   0.001082616;
52           gravc.j3     =  -0.00000253881;
53           gravc.j4     =  -0.00000165597;
54           gravc.j3oj2  =  gravc.j3 / gravc.j2;
55          case 84
56           % ------------ wgs-84 constants ------------
57           gravc.mu     = 398600.5;            %// in km3 / s2
58           gravc.radiusearthkm = 6378.137;     %// km
59           gravc.xke    = 60.0 / sqrt(gravc.radiusearthkm*gravc.radiusearthkm*gravc.radiusearthkm/gravc.mu);
60           gravc.tumin  = 1.0 / gravc.xke;
61           gravc.j2     =   0.00108262998905;
62           gravc.j3     =  -0.00000253215306;
63           gravc.j4     =  -0.00000161098761;
64           gravc.j3oj2  =  gravc.j3 / gravc.j2;
65          otherwise
66           fprintf('unknown gravity option (%d)\n',whichconst);
67      end;  % case
68
