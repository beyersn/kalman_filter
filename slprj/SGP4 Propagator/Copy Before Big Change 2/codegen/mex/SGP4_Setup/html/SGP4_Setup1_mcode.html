1function SGP4_Setup(longstr1, longstr2)
  2% SGP4_Setup
  3% Brandon Jackson
  4% bajackso@mtu.edu
  5% 9th July 2013
  6%
  7%
  8% Inputs:
  9% longstr1: TLE character String Line 1
 10% longstr2: TLE character Sring Line 2
 11%
 12% Outputs:
 13% satrec: structure containing all of the SGP4 satellite information 
 14%
 15% Coupling:
 16% getgravconst
 17% days2mdhms
 18% jday
 19% sgp4init
 20%
 21% References:
 22% Norad Spacetrack Report #3
 23% Vallado, Crawford, Hujsak, Kelso 2006
 24
 25%% Define Global Variables
 26global satrec gravc
 27
 28%% Include extrinsic functions
 29coder.extrinsic('str2num');
 30
 31%% WGS-72 Earth Constants
 32% sgp4fix identify constants and allow alternate values
 33% Options 721 72 84
 34getgravc( 72 );
 35
 36%% Define Constants
 37deg2rad  =   pi / 180.0;         %  0.01745329251994330;  % [deg/rad]
 38    xpdotp   =  1440.0 / (2.0*pi);   % 229.1831180523293;  % [rev/day]/[rad/min]  
 39
 40    revnum = 0; 
 41    elnum  = 0;
 42    year   = 0; 
 43    satrec.error = 0;
 44
 45%% Parse TLE
 46% Set the implied decimal points since doing a formated read fixes for bad
 47% input data values (missing, ...)
 48    for (j = 11:16)
 49        if (longstr1(j) == ' ')
 50            longstr1(j) = '_';
 51        end
 52    end
 53
 54    if (longstr1(45) ~= ' ')
 55        longstr1(44) = longstr1(45);
 56    end
 57    longstr1(45) = '.';
 58     
 59    if (longstr1(8) == ' ')
 60        longstr1(8) = 'U';
 61    end
 62
 63    if (longstr1(10) == ' ')
 64        longstr1(10) = '.';
 65    end
 66
 67    for (j = 46:50)
 68        if (longstr1(j) == ' ')
 69            longstr1(j) = '0';
 70        end
 71    end
 72    if (longstr1(52) == ' ')
 73        longstr1(52) = '0';
 74    end
 75    if (longstr1(54) ~= ' ')
 76        longstr1(53) = longstr1(54);
 77    end
 78    longstr1(54) = '.';
 79
 80    longstr2(26) = '.';
 81     
 82    for (j = 27:33)
 83        if (longstr2(j) == ' ')
 84            longstr2(j) = '0';
 85        end
 86    end
 87     
 88    if (longstr1(63) == ' ')
 89        longstr1(63) = '0';
 90    end
 91
 92    if ((length(longstr1) < 68) || (longstr1(68) == ' '))
 93        longstr1(68) = '0';
 94    end
 95
 96    % parse first line
 97    carnumb = str2num(longstr1(1));
 98    satrec.satnum = str2num(longstr1(3:7));
 99    classification = longstr1(8);
100    intldesg = longstr1(10:17);
101    satrec.epochyr = str2num(longstr1(19:20));
102    satrec.epochdays = str2num(longstr1(21:32));
103    satrec.ndot = str2num(longstr1(34:43));
104    satrec.nddot = str2num(longstr1(44:50));
105    nexp = str2num(longstr1(51:52));
106    satrec.bstar = str2num(longstr1(53:59));
107    ibexp = str2num(longstr1(60:61));
108    numb = str2num(longstr1(63));
109    elnum = str2num(longstr1(65:68));
110 
111    % parse second line
112    cardnumb = str2num(longstr2(1));
113    satrec.satnum = str2num(longstr2(3:7));
114    satrec.inclo = str2num(longstr2(8:16));
115    satrec.nodeo = str2num(longstr2(17:25));
116    satrec.ecco = str2num(longstr2(26:33));
117    satrec.argpo = str2num(longstr2(34:42));
118    satrec.mo = str2num(longstr2(43:51));
119    satrec.no = str2num(longstr2(52:63));
120    revnum = str2num(longstr2(64:68));
121
122% find no, ndot, nddot
123    satrec.no   = satrec.no / xpdotp; %//* rad/min
124    satrec.nddot= satrec.nddot * 10.0^nexp;
125    satrec.bstar= satrec.bstar * 10.0^ibexp;
126
127% convert to sgp4 units
128    satrec.a    = (satrec.no*gravc.tumin)^(-2/3);                % [er]
129    satrec.ndot = satrec.ndot  / (xpdotp*1440.0);          % [rad/min^2]
130    satrec.nddot= satrec.nddot / (xpdotp*1440.0*1440);     % [rad/min^3]
131
132% find standard orbital elements
133    satrec.inclo = satrec.inclo  * deg2rad;
134    satrec.nodeo = satrec.nodeo * deg2rad;
135    satrec.argpo = satrec.argpo  * deg2rad;
136    satrec.mo    = satrec.mo     *deg2rad;
137
138    satrec.alta = satrec.a*(1.0 + satrec.ecco) - 1.0;
139    satrec.altp = satrec.a*(1.0 - satrec.ecco) - 1.0;
140
141%% Find SGP4 Epoch Time of element set
142%  Remember that sgp4 uses units of days from 0 jan 1950 (sgp4epoch) and
143%  minutes from the epoch (time)
144
145% Temp fix for years 1957-2056
146% Correct fix will occur when year is 7-digits in 21e
147     if (satrec.epochyr < 57)
148         year= satrec.epochyr + 2000;
149       else
150         year= satrec.epochyr + 1900;
151     end;
152
153     [mon,day,hr,minute,sec] = days2mdh ( year,satrec.epochdays );
154     satrec.jdsatepoch = jday( year,mon,day,hr,minute,sec );
155     
156 %% Initialize the orbit at SGP4 Epoch    
157 sgp4epoch = satrec.jdsatepoch - 2433281.5; % days since 0 Jan 1950
158 sgp4init(sgp4epoch);
